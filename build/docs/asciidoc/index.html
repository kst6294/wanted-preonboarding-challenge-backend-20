content[].id

상품 ID (PK)

content[].productName

상품 명

content[].price

상품 가격

content[].productStatus

상품 상태 (ON_STOCK, SOLD_OUT 등)

content[].seller

상품 등록자 이메일

페이지 크기

number

전체 상품 수

totalPages

전체 페이지 수

first

last

마지막 페이지 여부

{
  "data": {
    "content": [

        "price": 10000,

      },
      {
        "id": 2,
        "productName": "Example Product 2",
        "price": 15000,
        "productStatus": "ON_STOCK",
        "seller": "user2@example.com"
      }
    ],
    "size": 5,
    "number": 0,
    "totalElements": 2,
    "totalPages": 1,
    "first": true,
    "last": true
  },
  "response": {
    "status": 200,
    "message": "success"
